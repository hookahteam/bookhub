Документация по работе с сервером Bookhub (для разработчиков)

# 1. Запуск
# Для стягивания изменений
git pull 

# Для установки всех необходимых зависимостей
./scripts/setup.sh 

# Дебаг режим (файлы фронта собираются статически)
make debug
# Сборка под продакшн
make release


# 2. Работа с бд (для бекенда)
# Если бд еще нет, то c root создаем
sudo -u postgres createdb bookhub

# *Посмотреть существующие бд
sudo -u postgres psql -l

# Создаем пользователя для деплоя (если его еще нет)
sudo -u postgres createuser deploy

# переключаемся на postgres консоль
sudo -u postgres psql

# Даем права на владение базой (чтобы он мог создавать таблицы)
ALTER DATABASE OWNER TO deploy;
